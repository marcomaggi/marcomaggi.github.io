70.5.4 Path component parser functions

The following functions can be used to parse the path-abempty, path-absolute, path-noscheme, path-rootless and path-empty components; the relevant portion of the grammar is: 

     path-abempty  = *( "/" segment )
     path-absolute = "/" [ segment-nz *( "/" segment ) ]
     path-noscheme = segment-nz-nc *( "/" segment )
     path-rootless = segment-nz *( "/" segment )
     path-empty    = 0<pchar>

notice that all the path-* components appear in the grammar as last components of a hier-part or relative-part component, which in turn can be followed only by EOF, a query (starting with a ‘?’ byte) or a fragment (starting with a ‘#’ byte).

The following bindings are exported by the (nausicaa uri low) library; it is suggested to prefix them when importing this library. All the functions accept a port argument: it must be a binary input port drawing characters from a bytevector.

— Function: parse-path-empty port

Parse a path-empty component. Lookahead one byte from port: if it is EOF or a question mark or number–sign in ASCII encoding: return null; else return false.

In any case leave the port position where it was before the function call. 

          (import (rnrs)
            (prefix (nausicaa uri low) low.))
          
          (define (make-lexer-port S)
            (open-bytevector-input-port (low.to-bytevector S)))
          
          (low.parse-path-empty (make-lexer-port ""))
          ⇒ ()
          
          (let ((P (make-lexer-port "ciao")))
            (low.parse-path-empty P)      ⇒ #f
            (low.to-string
              (get-bytevector-some P)))   ⇒ "ciao"
          
          (let ((P (make-lexer-port "/ciao")))
            (low.parse-path-empty P)      ⇒ #f
            (low.to-string
              (get-bytevector-some P)))   ⇒ "/ciao"
          
          (let ((P (make-lexer-port "?query")))
            (low.parse-path-empty P)      ⇒ ()
            (low.to-string
              (get-bytevector-some P)))   ⇒ "?query"
          
          (let ((P (make-lexer-port "#fragment")))
            (low.parse-path-empty P)      ⇒ ()
            (low.to-string
              (get-bytevector-some P)))   ⇒ "#fragment"
— Function: parse-path-abempty port

Parse from port a, possibly empty, sequence of sequences: byte representing the slash character in ASCII encoding, segment component. Return a possibly empty list holding bytevectors representing the segments.

If successful: leave the port position to the byte after the last read byte; if an error occurs: rewind the port position to the one before this function call.

If an invalid percent–encoded sequence is read, an exception is raised with type &parser-error; the port position is rewind to the one before this function call. 

          (import (rnrs)
            (prefix (nausicaa uri low) low.))
          
          (define (make-lexer-port S)
            (open-bytevector-input-port (low.to-bytevector S)))
          
          (low.parse-path-abempty (make-lexer-port ""))   ⇒ ()
          
          (let ((P (make-lexer-port "?query")))
            (low.parse-path-abempty P)                ⇒ #f
            (low.to-string (get-bytevector-some P)))  ⇒ "?query"
          
          (let ((P (make-lexer-port "#fragment")))
            (low.parse-path-abempty P)                ⇒ #f
            (low.to-string (get-bytevector-some P)))  ⇒ "#fragment"
          
          (let ((P (make-lexer-port "/ciao")))
            (map low.to-string
              (low.parse-path-abempty P))         ⇒ ("ciao")
            (eof-object? (lookahead-u8 P)))       ⇒ #t
          
          (let ((P (make-lexer-port "/ciao?query")))
            (map low.to-string
              (low.parse-path-abempty P))             ⇒ ("ciao")
            (low.to-string (get-bytevector-some P)))  ⇒ "?query"
          
          (let ((P (make-lexer-port "/ciao#fragment")))
            (map low.to-string
              (low.parse-path-abempty P))             ⇒ ("ciao")
            (low.to-string (get-bytevector-some P)))  ⇒ "#fragment"
          
          (let ((P (make-lexer-port "/ciao/hello")))
            (map low.to-string
              (low.parse-path-abempty P))    ⇒ ("ciao" "hello")
            (eof-object? (lookahead-u8 P)))  ⇒ #t
          
          (let ((P (make-lexer-port "/ciao/")))
            (map low.to-string
              (low.parse-path-abempty P))    ⇒ ("ciao" "")
            (eof-object? (lookahead-u8 P)))  ⇒ #t
          
          (map low.to-string
            (low.parse-path-abempty (make-lexer-port "///")))
          ⇒ ("" "" "")
— Function: parse-path-absolute port

Parse from port a path-absolute component; it is like parse-path-abempty, but expects a slash character as first byte and a non–slash as second byte. Return a possibly empty list holding bytevectors representing the segments, or false.

If successful: leave the port position to the byte after the last read byte; if an error occurs: rewind the port position to the one before this function call.

Notice that a path-absolute can be just a slash character, with no segments attached. 

          (import (rnrs)
            (prefix (nausicaa uri low) low.))
          
          (define (make-lexer-port S)
            (open-bytevector-input-port (low.to-bytevector S)))
          
          (low.parse-path-absolute (make-lexer-port ""))
          ⇒ #f
          
          (let ((P (make-lexer-port "ciao")))
            (low.parse-path-absolute P)               ⇒ #f
            (low.to-string (get-bytevector-some P)))  ⇒ "ciao"
          
          (let ((P (make-lexer-port "/")))
            (map low.to-string
              (low.parse-path-absolute P))            ⇒ ("")
              (eof-object? (lookahead-u8 P)))         ⇒ #t
          
          (let ((P (make-lexer-port "//")))
            (low.parse-path-absolute P)               ⇒ #f
            (low.to-string (get-bytevector-some P)))  ⇒ "//"
          
          (let ((P (make-lexer-port "/ciao")))
            (map low.to-string
              (low.parse-path-absolute P))            ⇒ ("ciao")
            (eof-object? (lookahead-u8 P)))           ⇒ #t
          
          (let ((P (make-lexer-port "/ciao/salut")))
            (map low.to-string
              (low.parse-path-absolute P))   ⇒ ("ciao" "salut")
            (eof-object? (lookahead-u8 P)))  ⇒ #t
          
          (let ((P (make-lexer-port "/ciao/")))
            (map low.to-string
              (low.parse-path-absolute P))   ⇒ ("ciao" "")
            (eof-object? (lookahead-u8 P)))  ⇒ #t
          
          (let ((P (make-lexer-port "/?query")))
            (map low.to-string
              (low.parse-path-absolute P))            ⇒ ("")
            (low.to-string (get-bytevector-some P)))  ⇒ "?query"
          
          (let ((P (make-lexer-port "/ciao?query")))
            (map low.to-string
              (low.parse-path-absolute P))            ⇒ ("ciao")
            (low.to-string (get-bytevector-some P)))  ⇒ "?query"
          
          (let ((P (make-lexer-port "/#fragment")))
            (map low.to-string
              (low.parse-path-absolute P))            ⇒ ("")
            (low.to-string (get-bytevector-some P)))  ⇒ "#fragment"
          
          (let ((P (make-lexer-port "/ciao#fragment")))
            (map low.to-string
              (low.parse-path-absolute P))            ⇒ ("ciao")
            (low.to-string (get-bytevector-some P)))  ⇒ "#fragment"
— Function: parse-path-noscheme port

Parse from port a path-noscheme component. Return a non–empty list holding bytevectors representing the segments, or false.

If successful: leave the port position to the byte after the last read byte; if an error occurs: rewind the port position to the one before this function call.

Notice that a path-noscheme must not start with a slash character, and then it must have at least one non–empty segment component; the first segment must not contain a colon caracter. 

          (import (rnrs)
            (prefix (nausicaa uri low) low.))
          
          (define (make-lexer-port S)
            (open-bytevector-input-port (low.to-bytevector S)))
          
          (low.parse-path-noscheme (make-lexer-port "")) ⇒ #f
          
          (let ((P (make-lexer-port "ciao")))
            (map low.to-string
              (low.parse-path-noscheme P))   ⇒ ("ciao")
            (eof-object? (lookahead-u8 P)))  ⇒ #t
          
          (let ((P (make-lexer-port "/")))
            (low.parse-path-noscheme P)               ⇒ #f
            (low.to-string (get-bytevector-some P)))  ⇒ "/"
          
          (let ((P (make-lexer-port "/ciao")))
            (low.parse-path-noscheme P)               ⇒ #f
            (low.to-string (get-bytevector-some P)))  ⇒ "/"
          
          (let ((P (make-lexer-port "ciao/hello")))
            (map low.to-string
              (low.parse-path-noscheme P))   ⇒ ("ciao" "hello")
            (eof-object? (lookahead-u8 P)))  ⇒ #t
          
          (let ((P (make-lexer-port "ciao/")))
            (map low.to-string
              (low.parse-path-noscheme P))  ⇒ ("ciao" "")
            (eof-object? (lookahead-u8 P))) ⇒ #t
          
          (let ((P (make-lexer-port "ciao/he:llo")))
            (map low.to-string
             (low.parse-path-noscheme P))   ⇒ ("ciao" "he:llo")
            (eof-object? (lookahead-u8 P))) ⇒ #t
          
          (let ((P (make-lexer-port "ci:ao/hello")))
            (low.parse-path-noscheme P)     ⇒ #f
            (low.to-string
              (get-bytevector-some P)))     ⇒ "ci:ao/hello"
          
          (let ((P (make-lexer-port "?query")))
            (low.parse-path-noscheme P)     ⇒ #f
            (low.to-string
              (get-bytevector-some P)))     ⇒ "?query"
          
          (let ((P (make-lexer-port "hello?ciao")))
            (map low.to-string
             (low.parse-path-noscheme P))   ⇒ ("hello")
            (low.to-string
              (get-bytevector-some P)))     ⇒ "?ciao"
          
          (let ((P (make-lexer-port "#fragment")))
            (low.parse-path-noscheme P)     ⇒ #f
            (low.to-string
              (get-bytevector-some P)))     ⇒ "#fragment"
          
          (let ((P (make-lexer-port "hello#fragment")))
            (map low.to-string
              (low.parse-path-noscheme P))  ⇒ ("hello")
            (low.to-string
              (get-bytevector-some P)))     ⇒ "#fragment"
— Function: parse-path-rootless port

Parse from port a path-rootless component; it is like parse-path-abempty, but expects a non–slash character as first byte. Return a non–empty list holding bytevectors representing the segments, or false.

If successful: leave the port position to the byte after the last read byte; if an error occurs: rewind the port position to the one before this function call.

Notice that a path-rootless must not start with a slash character, and then it must have at least one non–empty segment component. 

          (import (rnrs)
            (prefix (nausicaa uri low) low.))
          
          (define (make-lexer-port S)
            (open-bytevector-input-port (low.to-bytevector S)))
          
          (low.parse-path-rootless (make-lexer-port ""))  ⇒ #f
          
          (let ((P (make-lexer-port "ciao")))
            (map low.to-string
              (low.parse-path-rootless P))   ⇒ ("ciao")
            (eof-object? (lookahead-u8 P)))  ⇒ #t
          
          (let ((P (make-lexer-port "/")))
            (low.parse-path-rootless P)      ⇒ #f
            (low.to-string
              (get-bytevector-some P)))      ⇒ "/"
          
          (let ((P (make-lexer-port "/ciao")))
            (low.parse-path-rootless P)      ⇒ #f
            (low.to-string
              (get-bytevector-some P)))      ⇒ "/ciao"
          
          (let ((P (make-lexer-port "ciao/hello")))
            (map low.to-string
              (low.parse-path-rootless P))   ⇒ ("ciao" "hello")

          (let ((P (make-lexer-port "ci:ao/hel:lo")))
            (map low.to-string
              (low.parse-path-rootless P))   ⇒ ("ci:ao" "hel:lo")
            (eof-object? (lookahead-u8 P)))  ⇒ #t
          
          (let ((P (make-lexer-port "?query")))

              (get-bytevector-some P))       ⇒ "?query"
          
          (let ((P (make-lexer-port "ciao?query")))
            (low.parse-path-rootless P)      ⇒ ("ciao")
            (low.to-string
              (get-bytevector-some P))       ⇒ "?query"

            (low.to-string
              (get-bytevector-some P))       ⇒ "#fragment"
          
          (let ((P (make-lexer-port "ciao#fragment")))
            (low.parse-path-rootless P)      ⇒ ("ciao")
            (low.to-string
              (get-bytevector-some P))       ⇒ "#fragment"
