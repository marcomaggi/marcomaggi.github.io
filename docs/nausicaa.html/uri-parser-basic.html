72.5.1 Basic component parser functions

The following functions can be used to parse the elements of an URI; all the functions accept a port argument: it must be a binary input port drawing characters from an URI bytevector.

The following bindings are exported by the (nausicaa uri low) library; it is suggested to prefix them when importing this library.

— Function: parse-scheme port

Accumulate bytes from port while they are valid for the scheme component. If a colon is found: return a bytevector holding the accumulated bytes, colon excluded; else return false.

When successful: leave the port position to the byte after the one representing the colon; if an error occurs: rewind the port position to the one before this function call. 

          (import (rnrs)
            (prefix (nausicaa uri low) low.))
          
          (define (make-lexer-port S)
            (open-bytevector-input-port (low.to-bytevector S)))
          
          (low.parse-scheme (make-lexer-port ""))         ⇒ #f
          (low.parse-scheme (make-lexer-port "hello"))    ⇒ #f
          (low.parse-scheme (make-lexer-port "hel/lo:"))  ⇒ #f
          
          (let ((P (make-lexer-port "http://ciao")))
            (low.to-string (low.parse-scheme P))      ⇒ "http"
            (low.to-string (get-bytevector-some P)))  ⇒ "//ciao"
          
          (let ((P (make-lexer-port "A123+-.://ciao")))
            (low.to-string (low.parse-scheme P))      ⇒ "A123+-."
            (low.to-string (get-bytevector-some P)))  ⇒ "//ciao"
— Function: collect-hier-part port
— Function: collect-relative-part port

Accumulate bytes from port while they are acceptable for a hier-part component. If EOF or a question mark or a number–sign is found: return a bytevector holding the accumulated bytes, question mark or number–sign excluded; else return false.

Leave the port position to the byte after the last byte of the hier-part. An empty hier-part is not accepted: if the first value from the port is EOF, the return value is false.

No validation is performed on the returned bytevector. 

          (import (rnrs)
            (prefix (nausicaa uri low) low.))
          
          (define (make-lexer-port S)
            (open-bytevector-input-port (low.to-bytevector S)))
          
          (low.collect-hier-part (make-lexer-port ""))    ⇒ #f
          
          (let ((P (make-lexer-port "//")))
            (low.to-string (low.collect-hier-part P))  ⇒ "//"
            (eof-object? (lookahead-u8 P)))            ⇒ #t
          
          (let ((P (make-lexer-port "//ciao")))
            (low.to-string (low.collect-hier-part P))  ⇒ "//ciao"
            (eof-object? (lookahead-u8 P)))            ⇒ #t
          
          (let ((P (make-lexer-port "//ciao/salut")))
            (low.to-string (low.collect-hier-part P))  ⇒ "//ciao/salut"
            (eof-object? (lookahead-u8 P)))            ⇒ #f
          
          (let ((P (make-lexer-port "//ciao?query")))
            (low.to-string (low.collect-hier-part P))  ⇒ "//ciao"
            (low.to-string (get-bytevector-some P)))   ⇒ "?query"
          
          (let ((P (make-lexer-port "//ciao/salut?query")))
            (low.to-string (low.collect-hier-part P))  ⇒ "//ciao/salut"
            (low.to-string (get-bytevector-some P)))   ⇒ "?query"
          
          (let ((P (make-lexer-port "//ciao#fragment")))
            (low.to-string (low.collect-hier-part P))  ⇒ "//ciao"
            (low.to-string (get-bytevector-some P)))   ⇒ "#fragment"
          
          (let ((P (make-lexer-port "//ciao/salut#fragment")))
            (low.to-string (low.collect-hier-part P))  ⇒ "//ciao/salut"
            (low.to-string (get-bytevector-some P)))   ⇒ "#fragment"
          
          (let ((P (make-lexer-port "/")))
            (low.to-string (low.collect-hier-part P))  ⇒ "/"
            (eof-object? (lookahead-u8 P)))            ⇒ #t
          
          (let ((P (make-lexer-port "/ciao")))
            (low.to-string (low.collect-hier-part P))  ⇒ "/ciao"
            (eof-object? (lookahead-u8 P)))            ⇒ #t
          
          (let ((P (make-lexer-port "/ciao/salut")))
            (low.to-string (low.collect-hier-part P))  ⇒ "/ciao/salut"
            (eof-object? (lookahead-u8 P)))            ⇒ #t
          
          (let ((P (make-lexer-port "/ciao?query")))
            (low.to-string (low.collect-hier-part P))  ⇒ "/ciao"
            (low.to-string (get-bytevector-some P)))   ⇒ "?query"
          
          (let ((P (make-lexer-port "/ciao/salut?query")))
            (low.to-string (low.collect-hier-part P))  ⇒ "/ciao/salut"
            (low.to-string (get-bytevector-some P)))   ⇒ "?query"
          
          (let ((P (make-lexer-port "/ciao#fragment")))
            (low.to-string (low.collect-hier-part P))  ⇒ "/ciao"
            (low.to-string (get-bytevector-some P)))   ⇒ "#fragment"
          
          (let ((P (make-lexer-port "/ciao/salut#fragment")))
            (low.to-string (low.collect-hier-part P))  ⇒ "/ciao/salut"
            (low.to-string (get-bytevector-some P)))   ⇒ "#fragment"
          
          (let ((P (make-lexer-port ".")))
            (low.to-string (low.collect-hier-part P))  ⇒ "."
            (eof-object? (lookahead-u8 P)))            ⇒ #¯
          
          (let ((P (make-lexer-port "ciao")))
            (low.to-string (low.collect-hier-part P))  ⇒ "ciao"
            (eof-object? (lookahead-u8 P)))            ⇒ #t
          
          (let ((P (make-lexer-port "ciao/salut")))
            (low.to-string (low.collect-hier-part P))  ⇒ "ciao/salut"
            (eof-object? (lookahead-u8 P)))            ⇒ #t
          
          (let ((P (make-lexer-port "ciao?query")))
            (low.to-string (low.collect-hier-part P))  ⇒ "ciao"
            (low.to-string (get-bytevector-some P)))   ⇒ "?query"
          
          (let ((P (make-lexer-port "ciao/salut?query")))
            (low.to-string (low.collect-hier-part P))  ⇒ "ciao/salut"
            (low.to-string (get-bytevector-some P)))   ⇒ "?query"
          
          (let ((P (make-lexer-port "ciao#fragment")))
            (low.to-string (low.collect-hier-part P))  ⇒ "ciao"
            (low.to-string (get-bytevector-some P)))   ⇒ "#fragment"
          
          (let ((P (make-lexer-port "ciao/salut#fragment")))
            (low.to-string (low.collect-hier-part P))  ⇒ "ciao/salut"
            (low.to-string (get-bytevector-some P)))   ⇒ "#fragment"
— Function: parse-query port

Accumulate bytes from port while they are valid for a query URI component; the first byte read from port must be a question mark. If EOF or a number–sign (‘#’) is read: return a bytevector holding the accumulated bytes, starting question mark excluded and ending number–sign excluded; else return false.

If successful: leave the port position to the byte after the last byte of the query component; if an error occurs: rewind the port position to the one before this function call.

Notice that an empty query component is valid (a question mark followed by EOF). 

          (import (rnrs)
            (prefix (nausicaa uri low) low.))
          
          (define (make-lexer-port S)
            (open-bytevector-input-port (low.to-bytevector S)))
          
          (low.parse-query (make-lexer-port ""))          ⇒ #f
          (low.parse-query (make-lexer-port "hello"))     ⇒ #f
          (low.parse-query (make-lexer-port "#hello"))    ⇒ #f
          
          (low.to-string
           (low.parse-query (make-lexer-port "?")))
          ⇒ ""
          
          (low.to-string
           (low.parse-query (make-lexer-port "?the-query???")))
          ⇒ "the-query???"
          
          (let ((P (make-lexer-port "?ciao%3dciao#fragment")))
            (low.to-string (low.parse-query P))       ⇒ "ciao%3dciao"
            (low.to-string (get-bytevector-some P)))  ⇒ "#fragment"
— Function: parse-fragment port

Accumulate bytes from port while they are valid for a fragment component; the first byte read from port must be a number–sign. If EOF is read: return a bytevector holding the accumulated bytes, starting number–sign excluded; else return false.

If successful: leave the port position to the byte after the last byte of the fragment component; if an error occurs: rewind the port position to the one before this function call.

Notice that an empty fragment component is valid (a number–sign followed by EOF). 

          (import (rnrs)
            (prefix (nausicaa uri low) low.))
          
          (define (make-lexer-port S)
            (open-bytevector-input-port (low.to-bytevector S)))
          
          (low.parse-fragment (make-lexer-port ""))         ⇒ #f
          (low.parse-fragment (make-lexer-port "#hello#"))  ⇒ #f
          (low.parse-fragment (make-lexer-port "hello"))    ⇒ #f
          (low.parse-fragment (make-lexer-port "?hello"))   ⇒ #f
          
          (low.to-string
           (low.parse-fragment (make-lexer-port "#")))
          ⇒ ""
          
          (low.to-string
           (low.parse-fragment (make-lexer-port "#fragment???")))
          ⇒ "fragment???"
          
          (low.to-string
           (low.parse-fragment (make-lexer-port "#ciao%3dciao")))
          ⇒ "ciao%3dciao"
